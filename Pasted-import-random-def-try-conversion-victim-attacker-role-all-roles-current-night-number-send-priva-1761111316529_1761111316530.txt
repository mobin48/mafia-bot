import random

def try_conversion(victim, attacker_role, all_roles, current_night_number, send_private):
    """
    بررسی احتمال تبدیل قربانی به گرگ یا زامبی
    victim: بازیکن هدف
    attacker_role: نقش حمله‌کننده (مثلاً گرگ آلفا یا زامبی)

    if not victim_role_info:
        return None

def finalize_pending_conversions(players, current_night_number):
    """
    تبدیل نهایی بازیکنانی که قبلاً گاز گرفته شده‌اند.
    """
    for player in players:
        if (
            player.get("pending_conversion")
            and player["pending_conversion"]["type"] == "wolf"
            and player["pending_conversion"]["night_count"] < current_night_number
            and player["alive"]
        ):
            player["team"] = "گرگ‌ها"
            player["role"] = "گرگینه"
            del player["pending_conversion"]
